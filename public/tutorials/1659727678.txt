bool a[100000];
bool isPrime[1001];
bool vis[1001];
void seive() {
    for (int i = 2; i <= 1000; i++) {
        if (vis[i])continue;

        vis[i] = true;
        isPrime[i] = true;
        for (int j = i + i; j <= 1000; j += i) {
            if (vis[j])continue;
            else vis[j] = true;
        }
    }
}
